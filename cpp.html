C++ Basics Guide

A beginner's guide to understanding C++ programming concepts.

Syntax Variables Functions Conditionals Loops Classes

Syntax

C++ programs consist of functions, variables, and statements.

#include <iostream>
using namespace std;

int main() {
    cout << "Hello, World!" << endl;
    return 0;
}

Variables

Variables store data values in C++.

int age = 25;     // Integer variable
float pi = 3.14;  // Floating-point variable
char grade = 'A'; // Character variable
string name = "Alice"; // String variable

cout << "Age: " << age << endl;
cout << "PI: " << pi << endl;
cout << "Grade: " << grade << endl;
cout << "Name: " << name << endl;

Functions

Functions encapsulate reusable blocks of code in C++.

#include <iostream>
using namespace std;

int add(int a, int b) {
    return a + b;
}

int main() {
    cout << "Sum: " << add(5, 3) << endl;
    return 0;
}

Conditionals

Conditionals execute different code based on conditions.

#include <iostream>
using namespace std;

int main() {
    int age = 18;

    if (age >= 18) {
        cout << "You are an adult." << endl;
    } else {
        cout << "You are a minor." << endl;
    }

    return 0;
}

Loops

Loops execute a block of code multiple times.

#include <iostream>
using namespace std;

int main() {
    for (int i = 1; i <= 5; i++) {
        cout << "Count: " << i << endl;
    }

    return 0;
}

Classes

Classes define objects with properties and methods.

#include <iostream>
using namespace std;

class Car {
    public:
        string brand;
        int year;

        void display() {
            cout << brand << " " << year << endl;
        }
};

int main() {
    Car car1;
    car1.brand = "Toyota";
    car1.year = 2020;
    car1.display();

    return 0;
}
